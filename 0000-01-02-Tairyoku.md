function deleteNames(arr) {
    const unwantedNames = ['Дима', 'Саша', 'Ольга', 'Никита'];
    for (let unwantedName of unwantedNames) {
        for (let i = 0; i < arr.length ; ++i) {         
            if (arr[i] == unwantedName) {
                arr.splice(i, 1);
            }
        }
    }
    return arr;
}
let arr =['Алексей', 'Семён', 'Василиса', 'Дима', 'Максим', 'Ольга'];
alert(deleteNames(arr));
